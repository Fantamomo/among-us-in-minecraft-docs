import { Badge, CardGrid, LinkCard } from '@astrojs/starlight/components';

This plugin features a variety of roles, each with unique abilities and objectives.

## Crewmates

<CardGrid>
    <LinkCard title="Crewmate" href="../crewmates/crewmate" description="Standard crew member with no special abilities." />
    <LinkCard title="Caller" href="../crewmates/caller" description="Can call meetings from anywhere using a portable meeting button." />
    <LinkCard title="Camera Man" href="../crewmates/camera-man" description="Can access all cameras from anywhere on the map." />
    <LinkCard title="Detective" href="../crewmates/detective" description="Can track other players by their footprints." />
    <LinkCard title="Engineer" href="../crewmates/engineer" description="Can use vents like an Impostor." />
    <LinkCard title="Mayor" href="../crewmates/mayor" description="Has two votes in meetings instead of one." />
    <LinkCard title="Seer" href="../crewmates/seer" description="Can reveal the team of a selected player." />
    <LinkCard title="Sheriff" href="../crewmates/sheriff" description="Can kill players, but dies instead if the target is a Crewmate." />
    <LinkCard title="Snitch" href="../crewmates/snitch" description="Sees the Impostors once all tasks are completed." />
    <LinkCard title="The Damned" href="../crewmates/the-damned" description="No special abilities, but any killer who strikes them auto-reports." />
</CardGrid>

## Imposters

<CardGrid>
    <LinkCard title="Imposter" href="../imposters/imposter" description="Can kill Crewmates and sabotage ship systems." />
    <LinkCard title="Camouflager" href="../imposters/camouflager" description="Can disguise all players to look like a single random player." />
    <LinkCard title="Miner" href="../imposters/miner" description="Can create personal vent networks anywhere on the map." />
    <LinkCard title="Morphling" href="../imposters/morphling" description="Can morph into other players, copying their name, skin, and helmet." />
    <LinkCard title="Phantom" href="../imposters/phantom" description="Can enter ghost mode to move freely and kill while invisible." />
</CardGrid>
